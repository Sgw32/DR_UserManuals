%% Inline code formatting with breakable underscores
\ExplSyntaxOn
\cs_new_protected:Npn \dr_inline_code:n #1
  {
    \tl_set:Nn \l_tmpa_tl { #1 }
    \tl_replace_all:Nnn \l_tmpa_tl { \_ } { \_\allowbreak }
    \tl_use:N \l_tmpa_tl
  }
\NewDocumentCommand{\InlineCode}{m}
  {\texttt{\dr_inline_code:n { #1 }}}
\ExplSyntaxOff
